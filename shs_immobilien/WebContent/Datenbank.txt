CREATE TABLE objekte
    (id int primary key auto_increment,
    haustyp varchar(64),
    bautyp varchar(64),
    titel varchar(100),
    baujahr int(4),
    wohnflaeche int (4),
    grundstuecksflaeche int(6),
    standort varchar(32),
    datum date,
    gebot int(9),
    beschreibung varchar(500),
    bilder mediumblob,
    besitzer int(8),
    hoechstbietender int(8)
    );
    
    
CREATE TABLE haustyp
    (id int primary key auto_increment,
    typ varchar(64)
    );
    
    
CREATE TABLE bautyp
    (id int primary key auto_increment,
    typ varchar(64)
    );
    
    
CREATE TABLE benutzer 
    (id int primary key auto_increment,
    vorname varchar(32),
    nachname varchar(32),
    anschrift varchar(64),
    plz varchar(5),
    wohnort varchar(32),
    telefon varchar(16),
    email varchar(64),
    passwort1 varchar(32)
    );
