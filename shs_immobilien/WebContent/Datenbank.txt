create table objekte
    (id int primary key auto_increment,
    haustyp varchar(30),
    bautyp varchar(30),
    titel varchar(100),
    baujahr int(4),
    wohnflaeche int (5),
    grundstuecksflaeche int(6),
    standort varchar(100),
    datum date,
    startgebot int(10),
    beschreibung varchar(500),
    dateiname varchar(100),
    bilder mediumblob
    );
    
    
    create table benutzer
    (loginname varchar(80) primary key,
    passwort varchar(128)
    );
    
    create table haustyp
    (id int primary key auto_increment,
    typ varchar(255)
    );
    
    create table bautyp
    (id int primary key auto_increment,
    typ varchar(255)
    );
    
    
    create table benutzer 
    (id int primary key auto_increment,
    vorname varchar(30),
    nachname varchar(30),
    anschrift varchar(50),
    plz int(10),
    wohnort varchar(30),
    telefon int(20),
    email varchar(50),
    passwort1 varchar(20)
    );
