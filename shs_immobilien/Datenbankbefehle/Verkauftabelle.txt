create table objekte
    (id int primary key auto_increment,
    haustyp varchar(30),
    bautyp varchar(30),
    auktionsdauer date,
    titel varchar(100),
    baujahr int(4),
    wohnflaeche int (5),
    grundstuecksflaeche int(6),
    standort varchar(100),
    startgebot int(10),
    beschreibung varchar(500),
    dateiname varchar(100),
    bilder mediumblob
    );
